return "Invalid ID"

else:
if s[0] not in table:
return "Invalid ID"
else:
if s[1] not in ['1', '2']:
return "Invalid ID"
 1.檢查第一碼是否為table中的英文字母

else: tendigit=table[s[0]] // 10
